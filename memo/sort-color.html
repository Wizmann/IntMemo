Sort Color

Date 2015-03-24

difficulty 2      categories sort / leetcode      Task ID sort-color

https://leetcode.com/problems/sort-colors/

class Solution {
public:
    void sortColors(int A[], int n) {
        int l = 0;
        int r = n - 1;

        for (int i = 0; i <= r; i = max(i, l)) {
            switch (A[i]) {
                case 0:  { swap(A[i], A[l++]); } break;
                case 2:  { swap(A[i], A[r--]); } break;
                default: { i++; }                break;
            }
        }
    }
};
